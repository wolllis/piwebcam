# Raspberry Pi Zero 2 W firmware config for UVC webcam gadget
# Target: BCM2710A1 (RP3A0), 512 MB RAM, USB OTG peripheral, IMX477 HQ Camera

# Standard Pi firmware (no MMAL; camera driven by kernel + libcamera)
start_file=start.elf
fixup_file=fixup.dat
kernel=zImage

# GPU memory: 128 MB reserved for VideoCore ISP (bcm2835-isp / libcamera)
# Pi Zero 2 W has 512 MB; gpu_mem_512 applies to this board.
gpu_mem_512=128

# Kernel-side IMX477 sensor driver (used by bcm2835-unicam-legacy + libcamera)
dtoverlay=imx477

# Enable USB OTG in peripheral mode
dtoverlay=dwc2,dr_mode=peripheral

# Disable Bluetooth (releases PL011 UART to GPIO 14/15)
dtoverlay=disable-bt

# Disable WiFi (save power, not needed for USB gadget)
dtoverlay=disable-wifi

# Enable UART for debug
enable_uart=1
