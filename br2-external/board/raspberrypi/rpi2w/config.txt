# Raspberry Pi Zero 2 W firmware config for UVC webcam gadget
# This file is installed by rpi-firmware package to BINARIES_DIR/rpi-firmware/config.txt

# Standard Pi firmware
start_file=start.elf
fixup_file=fixup.dat
kernel=zImage

# Disable overscan
disable_overscan=1

# GPU memory (for models with different RAM sizes)
gpu_mem_256=100
gpu_mem_512=100
gpu_mem_1024=100

# Camera interface: the sensor is described explicitly via dtoverlay=imx477
# below, so disable firmware auto-detection to avoid a conflicting overlay.
camera_auto_detect=0

# CMA size for camera buffers (matches kernel CONFIG_CMA_SIZE_MBYTES=256)
dtoverlay=vc4-kms-v3d,cma-256

# IMX477 camera sensor overlay
dtoverlay=imx477

# Enable USB OTG in peripheral mode
dtoverlay=dwc2,dr_mode=peripheral

# Disable Bluetooth (releases PL011 UART to GPIO 14/15)
dtoverlay=disable-bt

# Disable WiFi (save power, not needed for USB gadget)
dtoverlay=disable-wifi

# Enable UART for debug
enable_uart=1
